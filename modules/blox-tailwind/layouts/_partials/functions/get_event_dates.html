A publication date is not necessarily the same as an event date. */}} {{/* Function to get event start and end dates/times. */}} {{ $t1 := time .Date }}{{/* Start datetime. */}} {{ if .Params.event_start }}{{ $t1 = time .Params.event_start }}{{ end }} {{ $str := slice (time.Format site.Params.locale.date_format (time $t1)) }}{{/* Init return string with start date. */}} {{/* Show start time if not all day event. */}} {{ if not .Params.all_day }}{{ $str = $str | append (time.Format site.Params.locale.time_format (time $t1)) }}{{ end }} {{/* Show event end date if provided. */}} {{ if .Params.event_end }} {{ $str = $str | append "—" }} {{ $t2 := time .Params.event_end}}{{/* End datetime. */}} {{/* Show end date if it differs to start date. */}} {{ if not (eq $t1.Year $t2.Year | and (eq $t1.Month $t2.Month) | and (eq $t1.Day $t2.Day)) }} {{ $str = $str | append (time.Format site.Params.locale.date_format (time $t2)) }} {{ end }} {{/* Show end time if not all day event. */}} {{ if not .Params.all_day }}{{ $str = $str | append (time.Format site.Params.locale.time_format (time $t2)) }}{{ end }} {{ end }} {{ return ((delimit $str " ") | safeHTML) }}
